def initialize(format, decode_vars)
  @format = format

  @opcode_fields = []
  msb = @format.size
  @format.split("-").each do |e|
    if e.empty?
      msb -= 1
    else
      @opcode_fields << Field.new(e, (msb - e.size + 1)..msb)
      msb -= e.size
    end
  end

  @decode_variables = []
  decode_vars.each do |var|
    @decode_variables << DecodeVariable.new(self, var)
  end
end
