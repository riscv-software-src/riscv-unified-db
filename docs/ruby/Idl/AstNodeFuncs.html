def nodes
  return @nodes unless @nodes.nil?

  @nodes = []

  elements.each do |e|
    if e.is_a?(AstNode)
      @nodes << e
    else
      nodes_helper(e)
    end
  end

  @nodes
end

def print_ast(indent = 0, indent_size: 2, io: $stdout)
  io.puts "#{' ' * indent}#{self.class.name}:"
  nodes.each do |node|
    node.print_ast(indent + indent_size, indent_size:)
  end
end
